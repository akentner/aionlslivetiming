# Design decisions made in this plan:

D-FILT-1 (API shape): BUILDER-PATTERN query object — `state.filter().by_class('SP9').top(5).cars()`. Rationale (from research/SUMMARY.md line 9-14 + line 215-218): composes well with method chaining, AND-semantics are implicit (each method narrows the set), top-N feels natural as `.top(5)` not `state.cars(class_='SP9')[:5]`. ALSO expose convenience methods on RaceState that delegate: `state.cars(class_='SP9')` returns the same Filter — this gives consumers the FastF1-style method-on-cache option without forcing it.

D-FILT-2 (driver match): SUBSTRING, case-insensitive. NLS driver names are "M. Müller", "F. Schmidt" — a consumer asking for "Mueller" (no umlaut, ASCII) should match "Müller". Use `fnmatch` case-insensitive OR a simple `lower() in lower()` substring check. Substring chosen over regex (regex is a footgun for users; substring is what FastF1 uses).

D-FILT-3 (position range): INCLUSIVE bounds. `by_position(min=1, max=5)` = positions 1, 2, 3, 4, 5. None position means "no position data yet" — exclude by default. Add `include_unknown_position()` opt-in to include them.

D-FILT-4 (lap range): INCLUSIVE bounds on laps_completed. `by_lap(min=20)` = laps_completed >= 20.

D-FILT-5 (sector threshold): STRICT less-than. `sector_time_lt(sector=1, value_ms=35000)` = sector_bests[sector] is not None AND < 35000. Sector 1 of Nordschleife is around 60s in the dry, so this is a sensible default for testing.

D-FILT-6 (return type): LIST (not generator). ARCHITECTURE.md line 161 explicit: "Returns a list, not a generator — easier to test and serialize."

D-FILT-7 (AND composition): every method narrows the working set; methods are idempotent (calling by_class('SP9') twice has no extra effect); no method ORs.

**`src/aionlslivetiming/state/filter.py`** — Filter query object. Each filter method narrows the working set; calling `.cars()` materialises the result as a list of `CarState` ordered by position (None last):

```python
from __future__ import annotations
from collections.abc import Iterable
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from aionlslivetiming.state.car import CarState
    from aionlslivetiming.state.race_state import RaceState


class Filter:
    """Composable query object over RaceState.cars. AND-semantics across dimensions."""

    def __init__(self, state: RaceState) -> None:
        self._state = state
        self._class_eq: set[str] = set()
        self._starting_no_in: set[int] | None = None
        self._driver_substr_lower: list[str] = []
        self._pos_min: int | None = None
        self._pos_max: int | None = None
        self._include_unknown_position: bool = False
        self._lap_min: int | None = None
        self._lap_max: int | None = None
        self._sector_lt: tuple[int, int] | None = None  # (sector, value_ms) strict less-than
        self._top_n: int | None = None

    # ---- dimension setters (each returns self for chaining) ----
    def by_class(self, class_name: str) -> Filter:
        self._class_eq.add(class_name)
        return self

    def by_starting_no(self, value: int | Iterable[int]) -> Filter:
        if self._starting_no_in is None:
            self._starting_no_in = set()
        if isinstance(value, int):
            self._starting_no_in.add(value)
        else:
            self._starting_no_in.update(int(v) for v in value)
        return self

    def by_driver(self, substring: str) -> Filter:
        # Case-insensitive substring (D-FILT-2)
        self._driver_substr_lower.append(substring.lower())
        return self

    def by_position(self, *, min: int | None = None, max: int | None = None) -> Filter:
        self._pos_min = min
        self._pos_max = max
        return self

    def by_lap(self, *, min: int | None = None, max: int | None = None) -> Filter:
        self._lap_min = min
        self._lap_max = max
        return self

    def sector_time_lt(self, *, sector: int, value_ms: int) -> Filter:
        self._sector_lt = (sector, value_ms)
        return self

    def include_unknown_position(self) -> Filter:
        self._include_unknown_position = True
        return self

    def top(self, n: int) -> Filter:
        self._top_n = n
        return self

    # ---- terminal: materialise the result ----
    def cars(self) -> list[CarState]:
        result: list[CarState] = list(self._state.cars.values())

        if self._class_eq:
            result = [c for c in result if c.class_name in self._class_eq]
        if self._starting_no_in is not None:
            result = [c for c in result if c.starting_no in self._starting_no_in]
        if self._driver_substr_lower:
            def driver_match(c: CarState) -> bool:
                if c.driver is None:
                    return False
                d = c.driver.lower()
                return any(sub in d for sub in self._driver_substr_lower)
            result = [c for c in result if driver_match(c)]
        if self._pos_min is not None or self._pos_max is not None or not self._include_unknown_position:
            def pos_match(c: CarState) -> bool:
                if c.position is None:
                    return self._include_unknown_position
                if self._pos_min is not None and c.position < self._pos_min:
                    return False
                if self._pos_max is not None and c.position > self._pos_max:
                    return False
                return True
            result = [c for c in result if pos_match(c)]
        if self._lap_min is not None or self._lap_max is not None:
            def lap_match(c: CarState) -> bool:
                if self._lap_min is not None and c.laps_completed < self._lap_min:
                    return False
                if self._lap_max is not None and c.laps_completed > self._lap_max:
                    return False
                return True
            result = [c for c in result if lap_match(c)]
        if self._sector_lt is not None:
            sec, threshold = self._sector_lt
            def sector_match(c: CarState) -> bool:
                v = c.sector_bests.get(sec)
                return v is not None and v < threshold
            result = [c for c in result if sector_match(c)]

        # Order: by position ascending, None last (matches RaceState.standings() ordering)
        result.sort(key=lambda c: (c.position is None, c.position if c.position is not None else 0))

        if self._top_n is not None:
            result = result[: self._top_n]
        return result

    # ---- terminal convenience: returning starting_nos only (common Discord bot / dashboard need) ----
    def starting_nos(self) -> list[int]:
        return [c.starting_no for c in self.cars()]
```

def filter(self) -> Filter:
    """Return a composable Filter over this state's cars."""
    from aionlslivetiming.state.filter import Filter
