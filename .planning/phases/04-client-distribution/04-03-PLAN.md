    **Step 4: Create `examples/live_quickstart.py`** — live-mode walkthrough with --dry-run support:

    ```python
    """Live quickstart example.

    Connects to a live NLS race and prints a snapshot of cached state every 30 seconds.

    Usage:
        # Real run (needs network + a valid event_id):
        uv run python examples/live_quickstart.py YOUR_EVENT_ID

        # CI mode (no network; uses a stub transport):
        uv run python examples/live_quickstart.py --dry-run

    Expected output:
        Connected to event 20 (source=LIVE)
        Streaming messages...
        After 30s: 142 messages, 3 cars in state, track=GREEN
    """

    from __future__ import annotations
    import argparse
    import asyncio
    import sys

    from aionlslivetiming import NLSClient, Source

    __all__ = ["main"]


    async def _run_live(event_id: str, duration_s: float) -> int:
        async with NLSClient(event_id=event_id) as client:
            assert client.source == Source.LIVE
            count = 0
            try:
                end_at = asyncio.get_event_loop().time() + duration_s
                while asyncio.get_event_loop().time() < end_at:
                    # Drain for 30s windows
                    async def _drain():
                        nonlocal count
                        async for _ in client.messages():
                            count += 1
                    await asyncio.wait_for(_drain(), timeout=min(30.0, end_at - asyncio.get_event_loop().time()))
            except asyncio.TimeoutError:
                pass
            print(f"After {duration_s}s: {count} messages, "
                  f"{len(client.state.cars)} cars in state, "
                  f"track={client.state.track}")
        return 0


    async def _run_dry() -> int:
        """CI mode: drive NLSClient.from_replay against the bundled sample."""
        from pathlib import Path
        sample = Path(__file__).parent.parent / "tests" / "fixtures" / "example_messages.jsonl"
        async with NLSClient.from_replay(sample) as client:
            count = 0
            async for _ in client.messages():
                count += 1
            print(f"Dry run: {count} messages from sample, "
                  f"{len(client.state.cars)} cars in state, "
                  f"track={client.state.track}")
        return 0


    def main(argv: list[str] | None = None) -> int:
        parser = argparse.ArgumentParser(prog="live_quickstart", description=__doc__)
        parser.add_argument("event_id", nargs="?", default="YOUR_EVENT_ID",
                            help="NLS event id (e.g. '20'). Use --dry-run for CI mode.")
        parser.add_argument("--dry-run", action="store_true",
                            help="Run against the bundled sample JSONL instead of the live server.")
        parser.add_argument("--duration-s", type=float, default=30.0,
                            help="How long to stream (default 30s).")
        args = parser.parse_args(argv)
        if args.dry_run:
            return asyncio.run(_run_dry())
        return asyncio.run(_run_live(args.event_id, args.duration_s))


    if __name__ == "__main__":
        sys.exit(main())
    ```

    **Step 5: Create `examples/replay_offline.py`** — replay-mode walkthrough:

    ```python
    """Replay offline example.

    Loads a recorded JSONL log via the NLSClient public API and prints a summary
    of the cached state at end-of-stream.

    Usage:
        uv run python examples/replay_offline.py /path/to/recording.jsonl

    Expected output:
        Replay: 142 messages from /path/to/recording.jsonl (source=REPLAY)
        State freshness: FRESH
        Cars cached: 3
        Track: GREEN
    """

    from __future__ import annotations
    import argparse
    import asyncio
    import sys
    from pathlib import Path

    from aionlslivetiming import NLSClient, Source

    __all__ = ["main"]


    async def _run(path: Path) -> int:
        async with NLSClient.from_replay(path) as client:
            assert client.source == Source.REPLAY
            count = 0
            async for _ in client.messages():
                count += 1
            print(f"Replay: {count} messages from {path} (source={client.source.value})")
            print(f"State freshness: {client.state.freshness.value}")
            print(f"Cars cached: {len(client.state.cars)}")
            print(f"Track: {client.state.track}")
        return 0


    def main(argv: list[str] | None = None) -> int:
        parser = argparse.ArgumentParser(prog="replay_offline", description=__doc__)
        parser.add_argument("jsonl_path", type=Path, help="Path to a recorded JSONL file.")
        args = parser.parse_args(argv)
        if not args.jsonl_path.exists():
            print(f"File not found: {args.jsonl_path}", file=sys.stderr)
            return 2
        return asyncio.run(_run(args.jsonl_path))


    if __name__ == "__main__":
        sys.exit(main())
    ```

    ```python
    """Tests for the worked examples (DOC-05).

    Each test invokes the example's main() in-process and captures stdout.
    No network access — live_quickstart.py is tested via its --dry-run mode.
    """

    from __future__ import annotations
    import pathlib
    import sys

    import pytest

    # Add the examples/ directory to sys.path so the test can import them.
    ROOT = pathlib.Path(__file__).resolve().parent.parent
    EXAMPLES_DIR = ROOT / "examples"
    sys.path.insert(0, str(EXAMPLES_DIR))

    from live_quickstart import main as live_main
    from replay_offline import main as replay_main
    from filter_walkthrough import main as filter_main

    EXAMPLE_JSONL = ROOT / "tests" / "fixtures" / "example_messages.jsonl"


    def test_live_quickstart_dry_run(capsys):
        """--dry-run uses the bundled sample; no network."""
        rc = live_main(["--dry-run"])
        assert rc == 0
        out = capsys.readouterr().out
        assert "Dry run:" in out or "messages" in out

    def test_live_quickstart_help(capsys):
        with pytest.raises(SystemExit) as exc:
            live_main(["--help"])
        assert exc.value.code == 0
        out = capsys.readouterr().out
        assert "--dry-run" in out
        assert "--duration-s" in out

    def test_replay_offline_with_sample_jsonl(capsys):
        rc = replay_main([str(EXAMPLE_JSONL)])
        assert rc == 0
        out = capsys.readouterr().out
        assert "Replay:" in out
        assert str(EXAMPLE_JSONL) in out
        assert "source=REPLAY" in out

    def test_replay_offline_file_not_found(capsys):
        rc = replay_main(["/nonexistent/path.jsonl"])
        assert rc == 2
        err = capsys.readouterr().err
        assert "not found" in err.lower() or "No such file" in err

    def test_filter_walkthrough_runs_against_sample(capsys):
        rc = filter_main([])
        assert rc == 0
        out = capsys.readouterr().out
        assert "Filter walkthrough" in out
        assert "by_class" in out
        assert "by_starting_no" in out
        assert "by_driver" in out
        assert "by_position" in out
        assert "by_lap" in out
        assert "by_sector_time_lt" in out

    def test_example_jsonl_is_valid_jsonl():
        """The sample JSONL must parse as one-JSON-object-per-line."""
        import json
        with EXAMPLE_JSONL.open() as f:
            lines = f.readlines()
        assert 5 <= len(lines) <= 20, f"expected 5-20 lines, got {len(lines)}"
        for i, line in enumerate(lines, 1):
            obj = json.loads(line)
            assert "ts_recv_ms" in obj, f"line {i} missing ts_recv_ms"
            assert "raw" in obj, f"line {i} missing raw"
        # Verify it contains at least PID 0, PID 4, PID 3
        pids = set()
        with EXAMPLE_JSONL.open() as f:
            for line in f:
                obj = json.loads(line)
                pid = obj["raw"].get("PID")
                if pid is not None:
                    pids.add(pid)
        assert 0 in pids, "sample_event.jsonl must contain a PID 0 (InitialState) message"
        assert 4 in pids, "sample_event.jsonl must contain a PID 4 (TrackState) message"
        assert 3 in pids, "sample_event.jsonl must contain a PID 3 (Race) message"
    ```
