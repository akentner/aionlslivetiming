            # Always closes the transport (which cancels reader task per Phase 3 D-08).

        async def messages(self) -> AsyncIterator[Message]:

        async def time_sync(self) -> AsyncIterator[TimeSyncMessage]:

                yield msg

            """Delegate to self._transport if it exposes lts_not_found(); else yield nothing."""
            lts = getattr(self._transport, "lts_not_found", None)
            if lts is None:
                return  # ReplayTransport does not expose this iterator
            async for ev in lts():
                yield ev

        # Accessors

            # Per agent discretion: "NLSClient(event_id=20, source=LIVE, state=<RaceState freshness=FRESH>)"
