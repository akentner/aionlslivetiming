    #!/usr/bin/env bash
    # scripts/check_no_ha_imports.sh
    # CI guard: fail if any homeassistant.* import appears in src/ or tests/.
    # Per Pitfall #14 + DIST-04 + D-30 — the core package must remain HA-free.

    set -euo pipefail

    SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
    ROOT="$(cd "${SCRIPT_DIR}/.." && pwd)"
    cd "${ROOT}"

    if grep -rn 'homeassistant' src/ tests/ 2>/dev/null | grep -v __pycache__ | grep -v '.pyc'; then
        echo "ERROR: homeassistant.* imports detected in src/ or tests/" >&2
        echo "The aionlslivetiming core package must not import from homeassistant.*" >&2
        echo "(HA integration is a separate concern — see PROJECT.md and v2 requirements HAI-01..03)" >&2
        exit 1
    fi

    **Step 5: Create `tests/test_build.py`** — automated verification (the key gates of this plan):

    ```python
    """Build verification tests (DIST-01, D-26..D-30).

    These tests verify that the package is publish-ready:
    - py.typed is included (PEP 561)
    - pyproject.toml metadata is complete
    - No homeassistant.* imports anywhere
    - uv build + twine check succeed (when run in an environment that has them)
    """

    from __future__ import annotations
    import pathlib
    import re
    import shutil
    import subprocess
    import zipfile

    import pytest

    ROOT = pathlib.Path(__file__).resolve().parent.parent


    def test_py_typed_marker_exists_in_src():
        """D-27 / D-11: py.typed must be a real file in src/."""
        assert (ROOT / "src" / "aionlslivetiming" / "py.typed").is_file(), \
            "src/aionlslivetiming/py.typed is missing — Phase 1 D-11 lock"

    @pytest.mark.parametrize("field_name", [
        "name", "version", "description", "readme", "license",
        "requires-python", "authors", "keywords", "classifiers",
    ])
    def test_pyproject_has_required_metadata_field(field_name: str):
        """D-28: every required PyPI metadata field is present in [project]."""
        pyproject = (ROOT / "pyproject.toml").read_text()
        # Look for `field_name =` or `field_name = {` inside [project] block
        # Simpler heuristic: just check the field name appears as a key
        assert re.search(rf"^{field_name}\s*=", pyproject, re.MULTILINE), \
            f"pyproject.toml missing required [project] field: {field_name}"

    def test_pyproject_has_project_urls():
        """D-28: Repository and Issues URLs are set."""
        pyproject = (ROOT / "pyproject.toml").read_text()
        assert "[project.urls]" in pyproject
        assert "Repository" in pyproject
        assert "Issues" in pyproject

    def test_pyproject_force_includes_py_typed():
        """D-11 / D-27: hatch force-include must include the py.typed file."""
        pyproject = (ROOT / "pyproject.toml").read_text()
        assert "force-include" in pyproject
        assert "py.typed" in pyproject

    def test_no_homeassistant_imports_in_src():
        """DIST-04 / D-30: zero homeassistant.* imports in src/."""
        result = list((ROOT / "src").rglob("*.py"))
        for path in result:
            text = path.read_text(encoding="utf-8", errors="replace")
            assert "homeassistant" not in text.lower(), \
                f"{path.relative_to(ROOT)} contains 'homeassistant' substring"

    def test_no_homeassistant_imports_in_tests():
        """D-30: zero homeassistant.* imports in tests/."""
        for path in (ROOT / "tests").rglob("*.py"):
            text = path.read_text(encoding="utf-8", errors="replace")
            assert "homeassistant" not in text.lower(), \
                f"{path.relative_to(ROOT)} contains 'homeassistant' substring"

    def test_check_no_ha_imports_script_executable():
        """The CI script must exist and be executable."""
        script = ROOT / "scripts" / "check_no_ha_imports.sh"
        assert script.is_file(), f"{script} missing"
        assert script.stat().st_mode & 0o111, f"{script} is not executable"

    def test_check_no_ha_imports_script_exits_zero():
        """Running the script exits 0 (no HA imports found)."""
        script = ROOT / "scripts" / "check_no_ha_imports.sh"
        result = subprocess.run(["bash", str(script)], capture_output=True, text=True)
        assert result.returncode == 0, \
            f"check_no_ha_imports.sh failed:\nSTDOUT: {result.stdout}\nSTDERR: {result.stderr}"

    def test_build_script_executable():
        """build.sh must exist and be executable."""
        script = ROOT / "scripts" / "build.sh"
        assert script.is_file(), f"{script} missing"
        assert script.stat().st_mode & 0o111, f"{script} is not executable"

    @pytest.mark.skipif(shutil.which("uv") is None, reason="uv not installed")
    def test_uv_build_produces_wheel_and_sdist(tmp_path):
        """D-26: uv build must produce both wheel and sdist."""
        result = subprocess.run(
            ["uv", "build", "--outdir", str(tmp_path)],
            cwd=ROOT, capture_output=True, text=True,
        )
        assert result.returncode == 0, \
            f"uv build failed:\nSTDOUT: {result.stdout}\nSTDERR: {result.stderr}"
        wheels = list(tmp_path.glob("*.whl"))
        sdists = list(tmp_path.glob("*.tar.gz"))
        assert len(wheels) == 1, f"expected 1 wheel, got {len(wheels)}: {wheels}"
        assert len(sdists) == 1, f"expected 1 sdist, got {len(sdists)}: {sdists}"

    @pytest.mark.skipif(shutil.which("uv") is None, reason="uv not installed")
    def test_wheel_contains_py_typed(tmp_path):
        """D-27: the built wheel must include py.typed."""
        build_dir = tmp_path / "build"
        build_dir.mkdir()
        result = subprocess.run(
            ["uv", "build", "--outdir", str(build_dir)],
            cwd=ROOT, capture_output=True, text=True,
        )
        assert result.returncode == 0
        wheel = next(build_dir.glob("*.whl"))
        with zipfile.ZipFile(wheel) as zf:
            names = zf.namelist()
        py_typed_paths = [n for n in names if n.endswith("py.typed")]
        assert py_typed_paths, \
            f"py.typed missing from wheel. Contents sample: {names[:10]}"

    @pytest.mark.skipif(shutil.which("uv") is None, reason="uv not installed")
    @pytest.mark.skipif(shutil.which("twine") is None and not _has_twine_module(),
                       reason="twine not installed")
    def test_twine_check_passes(tmp_path):
        """D-26: `python -m twine check dist/*` must pass."""
        build_dir = tmp_path / "build"
        build_dir.mkdir()
        subprocess.run(
            ["uv", "build", "--outdir", str(build_dir)],
            cwd=ROOT, capture_output=True, text=True, check=True,
        )
        result = subprocess.run(
            ["python", "-m", "twine", "check", *map(str, build_dir.iterdir())],
            capture_output=True, text=True,
        )
        assert result.returncode == 0, \
            f"twine check failed:\nSTDOUT: {result.stdout}\nSTDERR: {result.stderr}"


    def _has_twine_module() -> bool:
        """Helper: check if `twine` is importable as a Python module."""
        try:
            import twine  # noqa: F401
            return True
        except ImportError:
            return False
    ```
